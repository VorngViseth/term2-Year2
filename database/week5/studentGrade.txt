CREATE TABLE STUDENT_GRADES (
Student INT,
CourseNO INT,
Section INT,
Semester INT,
Year INT,
Grade INT,
    CONSTRAINT fk_student FOREIGN KEY (Student) REFERENCES STUDENT(SID),
    CONSTRAINT fk_section FOREIGN KEY (SecNum, Semester, Course)
        REFERENCES SECTION(SecNum, Semester, Course),
    CONSTRAINT fk_grade FOREIGN KEY (Grade) REFERENCES GRADE_REPORT(NumericGrade)
);
